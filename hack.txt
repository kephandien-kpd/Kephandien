-- chờ game load
repeat task.wait() until game:IsLoaded()
repeat task.wait() until game.Players.LocalPlayer

-- load thư viện
local Fluent = loadstring(game:HttpGet(
"https://raw.githubusercontent.com/dawid-scripts/Fluent/main/source.lua"))()

local SaveManager = loadstring(game:HttpGet(
"https://raw.githubusercontent.com/dawid-scripts/Fluent/main/Addons/SaveManager.lua"))()

local InterfaceManager = loadstring(game:HttpGet(
"https://raw.githubusercontent.com/dawid-scripts/Fluent/main/Addons/InterfaceManager.lua"))()

-- tạo cửa sổ
local Window = Fluent:CreateWindow({
    Title = "Kẻ Phản Diện Hub",
    SubTitle = "Menu Tiếng Việt",
    TabWidth = 160,
    Size = UDim2.fromOffset(520,380),
    Acrylic = true,
    Theme = "Dark",
    MinimizeKey = Enum.KeyCode.LeftControl
})

-- tab menu
local TrangChu = Window:AddTab({Title = "Trang Chủ"})
local Raid = Window:AddTab({Title = "Raid"})
local Farm = Window:AddTab({Title = "Farm"})
local NguoiChoi = Window:AddTab({Title = "Người Chơi"})
local CaiDat = Window:AddTab({Title = "Cài Đặt"})

-- nội dung
TrangChu:AddParagraph({
    Title = "Kẻ Phản Diện Hub",
    Content = "Menu đang hoạt động."
})

Raid:AddParagraph({
    Title = "Raid",
    Content = "Chức năng sẽ thêm sau."
})

Farm:AddParagraph({
    Title = "Farm",
    Content = "Chức năng sẽ thêm sau."
})

-- setup manager
SaveManager:SetLibrary(Fluent)
InterfaceManager:SetLibrary(Fluent)

InterfaceManager:BuildInterfaceSection(CaiDat)
SaveManager:BuildConfigSection(CaiDat)

-- khởi động UI
Fluent:Init()
