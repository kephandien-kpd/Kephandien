-- Toggle Logo Quạ (bản sửa lỗi)

local ScreenGui = Instance.new("ScreenGui")
ScreenGui.Name = "KPD_Toggle"
ScreenGui.Parent = game.CoreGui

local Toggle = Instance.new("ImageButton")
Toggle.Parent = ScreenGui
Toggle.Size = UDim2.new(0,70,0,70)
Toggle.Position = UDim2.new(0,20,0.5,0)
Toggle.BackgroundTransparency = 1
Toggle.Image = "rbxassetid://ID_ANH_QUA" -- thay id ảnh
Toggle.Active = true
Toggle.Draggable = true

-- đợi menu load
task.wait(3)

local Menu = nil
for _,v in pairs(game.CoreGui:GetChildren()) do
if v:IsA("ScreenGui") and v.Name ~= "KPD_Toggle" then
Menu = v
end
end

-- Toggle menu
Toggle.MouseButton1Click:Connect(function()
if Menu then
Menu.Enabled = not Menu.Enabled
end
end)
