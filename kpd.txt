local mouse = game.Players.LocalPlayer:GetMouse()

mouse.Button1Down:Connect(function()
    local target = mouse.Target
    if target then
        print("Đã chọn:", target:GetFullName())

        -- gán luôn làm quả cầu
        getgenv().RealBall = target
    end
end)

-- ===== HITBOX =====
getgenv().BallSize = 40

task.spawn(function()
    while true do
        if getgenv().RealBall then
            pcall(function()
                local v = getgenv().RealBall
                v.Size = Vector3.new(40,40,40)
                v.Transparency = 0.2
                v.Material = Enum.Material.Neon
                v.CanCollide = false
            end)
        end
        task.wait()
    end
end)
