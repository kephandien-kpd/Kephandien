import pyautogui
import time
import keyboard
import random

pyautogui.FAILSAFE = True

BUTTON_HIT = 'banana_hit.png'        
BUTTON_NEXT_ZONE = 'next_zone.png'    

def human_click(x, y):
    mx = x + random.randint(-5, 5)
    my = y + random.randint(-5, 5)
    pyautogui.moveTo(mx, my, duration=random.uniform(0.1, 0.3))
    pyautogui.click()
    time.sleep(random.uniform(0.05, 0.15))

def click_banana_style():
    for _ in range(random.randint(20, 40)):
        if keyboard.is_pressed('q'): 
            return False
        
        try:
            pos = pyautogui.locateOnScreen(BUTTON_HIT, confidence=0.8)
            if pos:
                center = pyautogui.center(pos)
                human_click(center.x, center.y)
        except Exception:
            pyautogui.click()
            time.sleep(random.uniform(0.01, 0.05))
    return True

def check_and_switch_zone():
    try:
        next_zone_pos = pyautogui.locateOnScreen(BUTTON_NEXT_ZONE, confidence=0.8)
        if next_zone_pos:
            center = pyautogui.center(next_zone_pos)
            human_click(center.x, center.y)
            time.sleep(3) 
            return True
    except Exception:
        pass
    return False

def main():
    time.sleep(3)

    while True:
        if keyboard.is_pressed('q'):
            break

        click_banana_style()
        check_and_switch_zone()
        time.sleep(random.uniform(0.5, 1.5))

if __name__ == "__main__":
    main()
