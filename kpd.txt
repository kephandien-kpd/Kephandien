-- Auto Checkout GUI (Safe)

local player = game.Players.LocalPlayer
local char = player.Character or player.CharacterAdded:Wait()

-- ===== GUI =====
local gui = Instance.new("ScreenGui", player:WaitForChild("PlayerGui"))
gui.Name = "AutoCheckoutGUI"

local frame = Instance.new("Frame", gui)
frame.Size = UDim2.new(0,200,0,120)
frame.Position = UDim2.new(0,20,0,200)
frame.BackgroundColor3 = Color3.fromRGB(30,30,30)
frame.Active = true
frame.Draggable = true

local toggle = Instance.new("TextButton", frame)
toggle.Size = UDim2.new(1,0,0,50)
toggle.Text = "Auto: OFF"
toggle.BackgroundColor3 = Color3.fromRGB(60,60,60)
toggle.TextColor3 = Color3.new(1,1,1)

local stop = Instance.new("TextButton", frame)
stop.Size = UDim2.new(1,0,0,50)
stop.Position = UDim2.new(0,0,0,60)
stop.Text = "STOP TP"
stop.BackgroundColor3 = Color3.fromRGB(100,40,40)
stop.TextColor3 = Color3.new(1,1,1)

-- ===== Logic =====
local auto = false
local allowTP = true

toggle.MouseButton1Click:Connect(function()
    auto = not auto
    toggle.Text = auto and "Auto: ON" or "Auto: OFF"
end)

stop.MouseButton1Click:Connect(function()
    allowTP = false
end)

-- Tìm quầy
local function findCheckout()
    for _, v in pairs(workspace:GetDescendants()) do
        if v.Name:lower():find("checkout") 
        or v.Name:lower():find("cashier") then
            return v
        end
    end
end

-- ===== Loop =====
task.spawn(function()
    while task.wait(1) do
        if not auto then continue end
        
        char = player.Character
        if not char or not char:FindFirstChild("HumanoidRootPart") then continue end
        
        local checkout = findCheckout()
        
        if checkout and allowTP then
            -- TP nhẹ tránh bug
            char.HumanoidRootPart.CFrame = char.HumanoidRootPart.CFrame:Lerp(
                checkout.CFrame + Vector3.new(0,3,0), 0.3
            )
        end
        
        -- Bấm thanh toán
        for _, v in pairs(workspace:GetDescendants()) do
            if v:IsA("ProximityPrompt") then
                fireproximityprompt(v)
            end
        end
    end
end)
