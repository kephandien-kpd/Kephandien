-- Auto Checkout Script

local player = game.Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()

-- Tìm quầy thanh toán (có thể cần chỉnh lại tên)
local function getCheckout()
    for _, v in pairs(workspace:GetDescendants()) do
        if v.Name:lower():find("checkout") or v.Name:lower():find("cashier") then
            return v
        end
    end
end

-- Auto thanh toán
while task.wait(1) do
    local checkout = getCheckout()
    
    if checkout and character and character:FindFirstChild("HumanoidRootPart") then
        -- Teleport đến quầy
        character.HumanoidRootPart.CFrame = checkout.CFrame + Vector3.new(0, 3, 0)
        
        -- Giả lập bấm thanh toán
        for _, v in pairs(checkout:GetDescendants()) do
            if v:IsA("ClickDetector") then
                fireclickdetector(v)
            end
            
            if v:IsA("ProximityPrompt") then
                fireproximityprompt(v)
            end
        end
    end
end
