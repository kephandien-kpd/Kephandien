local player = game.Players.LocalPlayer
local auto = false

-- ===== GUI =====
local gui = Instance.new("ScreenGui", player:WaitForChild("PlayerGui"))
gui.Name = "VIP_CHECKOUT"

local frame = Instance.new("Frame", gui)
frame.Size = UDim2.new(0,220,0,130)
frame.Position = UDim2.new(0,20,0,200)
frame.BackgroundColor3 = Color3.fromRGB(25,25,25)
frame.Active = true
frame.Draggable = true

local btn = Instance.new("TextButton", frame)
btn.Size = UDim2.new(1,0,0,50)
btn.Text = "AUTO: OFF"
btn.TextColor3 = Color3.new(1,1,1)
btn.BackgroundColor3 = Color3.fromRGB(50,50,50)

local info = Instance.new("TextLabel", frame)
info.Size = UDim2.new(1,0,0,70)
info.Position = UDim2.new(0,0,0,60)
info.Text = "Chờ quầy thanh toán..."
info.TextColor3 = Color3.new(1,1,1)
info.BackgroundTransparency = 1

btn.MouseButton1Click:Connect(function()
    auto = not auto
    btn.Text = auto and "AUTO: ON" or "AUTO: OFF"
end)

-- ===== Tìm tất cả tương tác =====
local function interactAll()
    for _, v in pairs(game:GetDescendants()) do
        
        -- ProximityPrompt
        if v:IsA("ProximityPrompt") then
            pcall(function()
                fireproximityprompt(v)
            end)
        end
        
        -- ClickDetector
        if v:IsA("ClickDetector") then
            pcall(function()
                fireclickdetector(v)
            end)
        end
        
        -- RemoteEvent (thử tất cả)
        if v:IsA("RemoteEvent") then
            pcall(function()
                v:FireServer()
            end)
        end
        
    end
end

-- ===== Loop chính =====
task.spawn(function()
    while task.wait(0.8) do
        if not auto then continue end
        
        local char = player.Character
        if not char or not char:FindFirstChild("HumanoidRootPart") then continue end
        
        -- Tìm quầy gần nhất
        local closest, dist = nil, math.huge
        
        for _, v in pairs(workspace:GetDescendants()) do
            if v:IsA("BasePart") and (
                v.Name:lower():find("cash") or 
                v.Name:lower():find("check") or 
                v.Name:lower():find("pay")
            ) then
                local d = (char.HumanoidRootPart.Position - v.Position).Magnitude
                if d < dist then
                    dist = d
                    closest = v
                end
            end
        end
        
        -- Di chuyển an toàn (không bị kẹt)
        if closest then
            info.Text = "Đang đến quầy..."
            char.Humanoid:MoveTo(closest.Position)
        end
        
        -- Thử thanh toán
        interactAll()
        info.Text = "Đang thanh toán..."
    end
end)
