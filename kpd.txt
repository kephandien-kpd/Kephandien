-- Auto Checkout (Advanced)

local player = game.Players.LocalPlayer

-- Tìm RemoteEvent liên quan thanh toán
local function findRemote()
    for _, v in pairs(game:GetDescendants()) do
        if v:IsA("RemoteEvent") then
            if v.Name:lower():find("pay") 
            or v.Name:lower():find("checkout") 
            or v.Name:lower():find("cash") then
                return v
            end
        end
    end
end

-- Tìm quầy thu ngân
local function findCheckout()
    for _, v in pairs(workspace:GetDescendants()) do
        if v.Name:lower():find("checkout") 
        or v.Name:lower():find("cashier") then
            return v
        end
    end
end

local remote = findRemote()

while task.wait(1) do
    local char = player.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then continue end

    local checkout = findCheckout()
    
    if checkout then
        -- Teleport tới quầy
        char.HumanoidRootPart.CFrame = checkout.CFrame + Vector3.new(0,3,0)
    end

    -- Nếu có remote thì spam thanh toán
    if remote then
        pcall(function()
            remote:FireServer()
        end)
    end
end
