getgenv().BallHitbox = true
getgenv().BallSize = 40

local function getBall()
    local bestBall = nil
    local maxSpeed = 0

    for _,v in pairs(workspace:GetDescendants()) do
        if v:IsA("BasePart") then
            local velocity = v.AssemblyLinearVelocity.Magnitude
            
            -- lọc vật di chuyển nhanh (giống quả cầu)
            if velocity > 20 and v.Size.X < 10 then
                if velocity > maxSpeed then
                    maxSpeed = velocity
                    bestBall = v
                end
            end
        end
    end

    return bestBall
end

task.spawn(function()
    while true do
        if getgenv().BallHitbox then
            local ball = getBall()
            if ball then
                pcall(function()
                    ball.Size = Vector3.new(getgenv().BallSize, getgenv().BallSize, getgenv().BallSize)
                    ball.Transparency = 0.2
                    ball.Material = Enum.Material.Neon
                    ball.CanCollide = false
                end)
            end
        end
        task.wait()
    end
end)
